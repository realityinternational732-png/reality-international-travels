  <script>
    // set year
    document.getElementById('year').textContent = new Date().getFullYear();

    // helper: save booking to localStorage
    function submitBooking(e){
      e.preventDefault();
      const f = e.target;
      const data = {
        id: Date.now(),
        name: f.name.value.trim(),
        email: f.email.value.trim(),
        phone: f.phone.value.trim(),
        destination: f.destination.value.trim(),
        date: f.date.value,
        message: f.message.value.trim()
      };
      const arr = JSON.parse(localStorage.getItem('ri_bookings')||'[]');
      arr.unshift(data);
      localStorage.setItem('ri_bookings', JSON.stringify(arr));
      f.reset();
      document.getElementById('formMsg').textContent = 'Booking submitted — we will contact you soon!';
      setTimeout(()=>document.getElementById('formMsg').textContent='',3500);
      return false;
    }

    // open booking modal prefilling destination
    function openBooking(title){
      document.getElementById('destinationInput').value = title;
      document.location = '#contact';
      window.scrollTo({top:document.getElementById('contact').offsetTop-20,behavior:'smooth'});
    }

    // quick preview (simple)
    function previewPackage(btn){
      const card = btn.closest('.package');
      const title = card.querySelector('.meta strong')?.textContent || 'Package';
      alert(title + "\nCheck booking section to reserve.");
    }

    // bookings dashboard
    document.getElementById('viewBookingsBtn').addEventListener('click', ()=>{
      const secret = prompt('Enter admin password to view bookings');
      // NOTE: this is client-side only. For production use a real backend.
      if(secret === 'admin123'){
        showBookings();
      } else if(secret !== null){
        alert('Wrong password');
      }
    });

    function showBookings(){
      const arr = JSON.parse(localStorage.getItem('ri_bookings')||'[]');
      const container = document.getElementById('bookingsContent');
      if(arr.length===0){
        container.innerHTML = '<div class="muted">No bookings yet.</div>';
      } else {
        let html = '<table><thead><tr><th>Name</th><th>Email</th><th>Phone</th><th>Destination</th><th>Date</th><th>Message</th><th>Actions</th></tr></thead><tbody>';
        for(const b of arr){
          html += `<tr data-id="${b.id}"><td>${escapeHtml(b.name)}</td><td>${escapeHtml(b.email)}</td><td>${escapeHtml(b.phone)}</td><td>${escapeHtml(b.destination)}</td><td>${escapeHtml(b.date)}</td><td>${escapeHtml(b.message)}</td><td><button class='btn ghost' onclick='deleteBooking(${b.id})'>Delete</button></td></tr>`;
        }
        html += '</tbody></table>';
        container.innerHTML = html;
      }
      document.getElementById('modal').classList.add('show');
      document.getElementById('modal').setAttribute('aria-hidden','false');
    }

    function closeModal(){
      document.getElementById('modal').classList.remove('show');
      document.getElementById('modal').setAttribute('aria-hidden','true');
    }

    function deleteBooking(id){
      if(!confirm('Delete this booking?')) return;
      let arr = JSON.parse(localStorage.getItem('ri_bookings')||'[]');
      arr = arr.filter(x=>x.id!==id);
      localStorage.setItem('ri_bookings', JSON.stringify(arr));
      showBookings();
    }

    function escapeHtml(str){
      if(!str) return '';
      return str.replace(/[&<>"']/g, function(m){return {'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":"&#39;"}[m]});
    }
